JS之倒计时

圣诞节就在眼前 一起来做个倒计时吧 (｡◕ˇ∀ˇ◕）

var timing = setInterval(function(){
    var fTime = new Date(2015,11,24,0,0,0);
    var nTime = new Date();
    var time = (fTime - nTime)/1000;
    var day = Math.floor(time/60/60/24);
    var hour = Math.floor(time/60/60-day*24);
    var min = Math.floor(time/60-day*24*60-hour*60);
    var sec = Math.floor(time-day*24*60*60-hour*60*60-min*60);
    var str = zero(day)+""+zero(hour)+""+zero(min)+""+zero(sec);

    for(var i = 0; i<timing.length;i++){
        pic[i].innerHTML = str.charAt(i);
    }

    if (time <= 0) {
        clearInterval(timing);
    };
},30)

function zero(num){
    if(num<10){
        return "0"+num;
    }else{
        return num;
    }
}
